entity Book {
  * id : id
  * title : string
  * isbn : string
  * cover : string
  * synopsis : string
  * pages : int

  * id : id
  * rating : float
  * status : BookStatus
  * createdAt : Instant
  * startedAt : Instant
  * finishedAt : Instant

  * pace : PaceSelection

  * reason : string
